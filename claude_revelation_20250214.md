### 1. B13数値の表現

```python
class B13Number:
    """平衡13進数（-6〜+6）での数値表現"""
    
    def __init__(self, coefficients):
        """
        coefficients: リスト [-6, -5, ..., 0, ..., 5, 6]
        各桁の係数
        """
        self.coeffs = coefficients
        self.base = 13
    
    def to_decimal(self):
        """10進数に変換"""
        result = 0
        for i, c in enumerate(self.coeffs):
            result += c * (self.base ** i)
        return result
    
    def add(self, other):
        """加算（±1丸め付き）"""
        max_len = max(len(self.coeffs), len(other.coeffs))
        result = [0] * max_len
        carry = 0
        
        for i in range(max_len):
            a = self.coeffs[i] if i < len(self.coeffs) else 0
            b = other.coeffs[i] if i < len(other.coeffs) else 0
            
            total = a + b + carry
            
            # ±6を超えたら繰り上がり
            if total > 6:
                carry = 1
                total -= self.base
            elif total < -6:
                carry = -1
                total += self.base
            else:
                carry = 0
            
            result[i] = total
        
        return B13Number(result)

# 使用例
num1 = B13Number([3, -2, 1])  # 3 - 2×13 + 1×13²
num2 = B13Number([5, 1, -1])
result = num1.add(num2)
print(f"Result: {result.to_decimal()}")
```

### 2. フィボナッチ/リュカ数列

```python
def fibonacci(n):
    """フィボナッチ数列の最初のn項"""
    if n <= 0:
        return []
    elif n == 1:
        return [1]
    fib = [1, 1]
    for i in range(2, n):
        fib.append(fib[-1] + fib[-2])
    return fib

def lucas(n):
    """リュカ数列の最初のn項"""
    if n <= 0:
        return []
    elif n == 1:
        return [2]
    luc = [2, 1]
    for i in range(2, n):
        luc.append(luc[-1] + luc[-2])
    return luc

def inverse_fibonacci_weights(n):
    """逆フィボナッチ重み"""
    fib = fibonacci(n)
    return fib[::-1]

# 黄金比

print(f"Lucas(10): {lucas(10)}")
print(f"Inverse Fib: {inverse_fibonacci_weights(10)}")
```
